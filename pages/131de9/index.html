学籍注册程序

# 1. 要求

  1. 数据存到文件的格式: 姓名、年龄、手机、身份证、学科。
  2. 要求确保手机、身份证的唯一性。可以手机和身份证加载到内存里，以列表形式存放。用户输入相关列，就在列表里检查是否重复。
  3. 选学科时，给用户列出选项供选择。

# 2.知识点

# 1.1. enumerate() 函数

描述

enumerate() 函数用于将一个可遍历的数据对象(如列表、元组或字符串)组合为一个索引序列，同时列出数据和数据下标，一般用在 for 循环当中。

语法

enumerate(sequence, [start=0])
1

实例

>>>seasons = ['Spring', 'Summer', 'Fall', 'Winter']
>>> list(enumerate(seasons))
[(0, 'Spring'), (1, 'Summer'), (2, 'Fall'), (3, 'Winter')]
>>> list(enumerate(seasons, start=1))       # 下标从 1 开始
[(1, 'Spring'), (2, 'Summer'), (3, 'Fall'), (4, 'Winter')]

# 3. 实现

db_file = "student_data.db"


def register_api():
    stu_data = {}  # 为了后边存学院数据
    print('欢迎来到学籍注册页面'.center(50, "-"))
    print('请完成学籍注册')
    name = input('姓名：').strip()
    age = input('年龄：').strip()
    phone = input('手机号：').strip()
    if phone in load_phone_list:
        exit("该手机号已注册")
    id_num = input('身份证号：').strip()
    if id_num in load_id_num_list:
        exit("该身份证号已注册")

    course_list = ['Python开发', 'Linux云计算', '网络安全', '数据分析', '前端开发']
    for index, course in enumerate(course_list):
        print(f"{index}.{course}")

    selected_course = input("选择想学的课程：")
    if selected_course.isdigit():
        selected_course = int(selected_course)
        if selected_course >= 0 and selected_course < len(course_list):  # 合法选项
            picked_course = course_list[selected_course]  # 选中的课程
        else:
            exit('不合法的选项...')
    else:
        exit('非法输入...')

    stu_data['name'] = name
    stu_data['age'] = age
    stu_data['phone'] = phone
    stu_data['id_num'] = id_num
    stu_data['course'] = picked_course

    return stu_data


def commit_to_db(filename, stu_data):
    """[存储学员数据]

    Args:
        filename ([type]): [学院数据库文件]
        stu_data ([type]): [单个学院数据的dict]
    """
    f = open(filename, "a")
    row = f"{stu_data['name']},{stu_data['age']},{stu_data['phone']},{stu_data['id_num']},{stu_data['course']}\n"
    f.write(row)
    f.close()


def load_validated_data(filename):
    f = open(filename)
    phone_list = []
    id_num_list = []
    for line in f:
        line = line.split(",")
        phone = line[2]
        id_num = line[3]
        phone_list.append(phone)
        id_num_list.append(id_num)

    return phone_list, id_num_list


load_phone_list, load_id_num_list = load_validated_data(db_file)

student_data = register_api()
print(student_data)
commit_to_db(db_file, student_data)
